7)app.py

from flask import Flask, render_template, request, redirect
import mysql.connector
from config import DB_CONFIG, SECRET_KEY

app = Flask(__name__)
app.secret_key = SECRET_KEY


# ---------------- HOME PAGE ----------------
@app.route("/")
def index():
    return render_template("index.html")


# ---------------- MENU PAGE (WITH SEASONAL FEATURE) ----------------
@app.route("/menu")
def menu():
    conn = mysql.connector.connect(**DB_CONFIG)
    cur = conn.cursor(dictionary=True)

    # Normal menu items
    cur.execute("SELECT * FROM menu_items WHERE is_seasonal = 'NO'")
    items = cur.fetchall()

    # Seasonal items (date-based)
    cur.execute("""
        SELECT mi.*, si.festival_name
        FROM menu_items mi
        JOIN seasonal_items si ON mi.item_id = si.item_id
        WHERE CURDATE() BETWEEN si.start_date AND si.end_date
    """)
    seasonal = cur.fetchall()

    cur.close()
    conn.close()

    return render_template("menu.html", items=items, seasonal=seasonal)


# ---------------- PLACE ORDER ----------------
@app.route("/place_order", methods=["POST"])
def place_order():
    conn = mysql.connector.connect(**DB_CONFIG)
    cur = conn.cursor(dictionary=True)

    # Create new order
    cur.execute("INSERT INTO orders(student_id) VALUES (1)")
    conn.commit()

    order_id = cur.lastrowid
    token = order_id + 100

    # Insert items
    for key, value in request.form.items():
        if key.startswith("item_") and int(value) > 0:
            item_id = int(key.split("_")[1])
            qty = int(value)
            cur.execute("INSERT INTO order_items(order_id, item_id, quantity) VALUES (%s, %s, %s)",
                        (order_id, item_id, qty))

    conn.commit()

    # Calculate bill
    cur.execute("""
        SELECT SUM(mi.price * oi.quantity) AS total
        FROM order_items oi
        JOIN menu_items mi ON oi.item_id = mi.item_id
        WHERE oi.order_id = %s
    """, (order_id,))
    total = cur.fetchone()["total"]

    # Update total in orders table
    cur.execute("UPDATE orders SET total_amount=%s, token_number=%s WHERE order_id=%s",
                (total, token, order_id))
    conn.commit()

    cur.close()
    conn.close()

    return render_template("order_success.html", token=token, order_id=order_id, total=total)


# ---------------- ADMIN PANEL ----------------
@app.route("/admin", methods=["GET", "POST"])
def admin():
    conn = mysql.connector.connect(**DB_CONFIG)
    cur = conn.cursor(dictionary=True)

    if request.method == "POST":
        item_id = request.form["item_id"]
        qty = int(request.form["qty"])
        cur.execute("UPDATE menu_items SET stock = stock + %s WHERE item_id = %s", (qty, item_id))
        conn.commit()

    cur.execute("SELECT * FROM menu_items")
    items = cur.fetchall()

    cur.close()
    conn.close()

    return render_template("admin.html", items=items)


# ---------------- REPORTS PAGE ----------------
@app.route("/reports")
def reports():
    conn = mysql.connector.connect(**DB_CONFIG)
    cur = conn.cursor(dictionary=True)

    cur.execute("""
        SELECT mi.item_name, SUM(oi.quantity) AS total_sold
        FROM order_items oi
        JOIN menu_items mi ON oi.item_id = mi.item_id
        GROUP BY oi.item_id
        ORDER BY total_sold DESC
    """)
    popular = cur.fetchall()

    cur.execute("""
        SELECT DATE(order_time) AS day, SUM(total_amount) AS total_sales
        FROM orders
        GROUP BY DATE(order_time)
        ORDER BY day DESC
    """)
    sales = cur.fetchall()

    cur.execute("""
        SELECT mi.item_name, et.expiry_date
        FROM expiry_tracker et
        JOIN menu_items mi ON et.item_id = mi.item_id
        WHERE et.is_expired = 'YES'
    """)
    expired = cur.fetchall()

    cur.close()
    conn.close()

    return render_template("reports.html", popular=popular, sales=sales, expired=expired)
